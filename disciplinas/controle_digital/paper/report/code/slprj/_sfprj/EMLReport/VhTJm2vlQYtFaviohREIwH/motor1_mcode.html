1function [T, i, y] = motor(u, w_m, params)
 2    % Motor parameters
 3    % Simplifications: 
 4    %   Negligible pulley inertia and stator inductance
 5    Ra = params.Ra;
 6    Kt = params.Kt;
 7    Ke = params.Ke;
 8    
 9    % Exerted Torque
10    T = (Kt/Ra)*(u - Ke*w_m);
11    
12    % Current
13    i = T/Kt;
14    
15    % Angular speed
16    y = w_m;
17end
